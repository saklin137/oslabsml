Selection Sort


Arr=(15 2 3 24)

length=${#Arr[@]}

echo "Before Sorting Array element are : ${Arr[@]}"

#Selection Sort Algorithm

for((i=0;i<length-1;i++))

do

minimumIndex=$i

for((j=i+1;j<length;j++))

do

if [ ${Arr[j]} -lt ${Arr[minimumIndex]} ]

then

minimumIndex=$j

fi

done

temp=${Arr[$i]}

Arr[$i]=${Arr[minimumIndex]}

Arr[$minimumIndex]=$temp

done

echo "After sorting your Array is ${Arr[@]}"
